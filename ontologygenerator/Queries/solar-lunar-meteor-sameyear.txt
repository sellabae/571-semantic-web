PREFIX owl:<http://www.w3.org/2002/07/owl#>
PREFIX time:<http://www.w3.org/2006/time#>
PREFIX rdf:<http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX base: <http://webprotege.stanford.edu/>
PREFIX geo: <http://www.w3.org/2003/01/geo/wgs84_pos#>
PREFIX nasa: <https://data.nasa.gov/Space-Science/Meteorite-Landings/gh4g-9sfh/>
PREFIX gml:<http://www.opengis.net/>

SELECT Distinct ?solarCatalogId ?lunarCatalogId ?meteorName
WHERE
{
        { ?solarEclipse a base:solarEclipse ; 
                    base:hasCatalogId ?solarCatalogId ;
					time:month ?solarMonth;
        			time:year ?solarYear.	
       
    }
    .
    {
    ?lunarEclipse a base:lunarEclipse;
                  base:hasCatalogId ?lunarCatalogId ;
                  time:month ?lunarMonth;
                  time:year ?lunarYear.
      
                  }
    .   
    {
        ?meteor a nasa:data;
                base:hasName ?meteorName;
                time:year ?meteorYear.
    }
    .Filter( 
        ?meteorYear=?lunarYear && ?lunarYear=?solarYear && ?meteorYear=?solarYear).    

}
